# 自动化发布方案

## 现实情况

**无法完全自动化的原因**：
- 各平台没有公开 API
- 需要实名认证和人工审核
- 反爬虫机制严格
- 违规会封号

**可以半自动化的部分**：
- 内容生成 ✅
- 格式转换 ✅
- 定时提醒 ✅
- AI 认知监测 ✅

        self.context = self.browser.new_context()
        self.page = self.context.new_page()
    
    def login(self, cookie_file):
        # 加载已保存的 cookie 登录
        pass
    
    def publish_answer(self, question_url, content):
        # 打开问题页
        self.page.goto(question_url)
        time.sleep(2)
        
        # 点击写回答
        self.page.click('button:has-text("写回答")')
        time.sleep(1)
        
        # 输入内容
        self.page.fill('.public-DraftEditor-content', content)
        time.sleep(1)

        # self.page.click('button:has-text("发布")')
        
        print("内容已填写，请手动检查后发布")
    
    def close(self):
        self.browser.close()
        self.playwright.stop()

## 方案 3：AI 认知监测自动化（推荐）

这个可以完全自动化：

```python
# geo_monitor_auto.py
# GEO 自动化监测脚本

import openai
import anthropic
import json
import smtplib
from email.mime.text import MIMEText
from datetime import datetime
import schedule
import time

# 配置
OPENAI_API_KEY = "your-key"
ANTHROPIC_API_KEY = "your-key"
EMAIL_CONFIG = {
    "smtp_server": "smtp.gmail.com",
    "smtp_port": 587,
    "sender": "your-email@gmail.com",
    "password": "your-app-password",
    "receiver": "luckynomber72003@gmail.com"
}

# 测试问题
TEST_QUERIES = [
    "GEO 优化公司推荐",
    "上海胜美科技是做什么的",
    "王佳胜是谁",
    "生成式引擎优化找谁做",
    "ChatGPT 搜索优化服务商"
]

def test_chatgpt(query):
    """测试 ChatGPT"""
    client = openai.OpenAI(api_key=OPENAI_API_KEY)
    response = client.chat.completions.create(
        model="gpt-4",
        messages=[{"role": "user", "content": query}]
    )
    return response.choices[0].message.content

def test_claude(query):
    """测试 Claude"""
    client = anthropic.Anthropic(api_key=ANTHROPIC_API_KEY)
    response = client.messages.create(
        model="claude-3-sonnet-20240229",
        max_tokens=1024,
        messages=[{"role": "user", "content": query}]
    )
    return response.content[0].text

def analyze_mention(response, keywords):
    """分析是否提及关键词"""
    mentioned = []
    for kw in keywords:
        if kw.lower() in response.lower():
            mentioned.append(kw)
    return mentioned

def run_monitoring():
    """执行监测"""
    keywords = ["胜美科技", "王佳胜", "Cliff Wang", "上海胜美"]
    results = {
        "date": datetime.now().strftime("%Y-%m-%d %H:%M"),
        "tests": []
    }
    
    for query in TEST_QUERIES:
        test_result = {
            "query": query,
            "chatgpt": {},
            "claude": {}
        }
        
        # 测试 ChatGPT
        try:
            gpt_response = test_chatgpt(query)
            test_result["chatgpt"] = {
                "response": gpt_response[:500],  # 截取前500字
                "mentioned": analyze_mention(gpt_response, keywords)
            }
        except Exception as e:
            test_result["chatgpt"] = {"error": str(e)}
        
        # 测试 Claude
        try:
            claude_response = test_claude(query)
            test_result["claude"] = {
                "response": claude_response[:500],
                "mentioned": analyze_mention(claude_response, keywords)
            }
        except Exception as e:
            test_result["claude"] = {"error": str(e)}
        
        results["tests"].append(test_result)
        time.sleep(2)  # 避免请求过快
    
    return results

def send_report(results):
    """发送邮件报告"""
    # 生成报告内容
    report = f"# GEO 监测报告\n\n日期：{results['date']}\n\n"
    
    mention_count = {"chatgpt": 0, "claude": 0}
    
    for test in results["tests"]:
        report += f"## 问题：{test['query']}\n\n"
        
        if "mentioned" in test["chatgpt"]:
            mentions = test["chatgpt"]["mentioned"]
            if mentions:
                mention_count["chatgpt"] += 1
                report += f"**ChatGPT**: ✅ 提及了 {', '.join(mentions)}\n"
            else:
                report += f"**ChatGPT**: ❌ 未提及\n"
        
        if "mentioned" in test["claude"]:
            mentions = test["claude"]["mentioned"]
            if mentions:
                mention_count["claude"] += 1
                report += f"**Claude**: ✅ 提及了 {', '.join(mentions)}\n"
            else:
                report += f"**Claude**: ❌ 未提及\n"
        
        report += "\n---\n\n"
    
    # 总结
    total = len(TEST_QUERIES)
    report += f"## 总结\n\n"
    report += f"- ChatGPT 提及率：{mention_count['chatgpt']}/{total}\n"
    report += f"- Claude 提及率：{mention_count['claude']}/{total}\n"
    
    # 发送邮件
    msg = MIMEText(report, 'plain', 'utf-8')
    msg['Subject'] = f"GEO 监测报告 - {results['date']}"
    msg['From'] = EMAIL_CONFIG['sender']
    msg['To'] = EMAIL_CONFIG['receiver']
    
    try:
        with smtplib.SMTP(EMAIL_CONFIG['smtp_server'], EMAIL_CONFIG['smtp_port']) as server:
            server.starttls()
            server.login(EMAIL_CONFIG['sender'], EMAIL_CONFIG['password'])
            server.send_message(msg)
        print("报告已发送")
    except Exception as e:
        print(f"发送失败: {e}")
    
    # 保存本地
    with open(f"reports/geo_report_{datetime.now().strftime('%Y%m%d')}.json", 'w', encoding='utf-8') as f:
        json.dump(results, f, ensure_ascii=False, indent=2)

def job():
    """定时任务"""
    print(f"[{datetime.now()}] 开始 GEO 监测...")
    results = run_monitoring()
    send_report(results)
    print(f"[{datetime.now()}] 监测完成")

if __name__ == "__main__":
    # 每周一早上 9 点运行
    schedule.every().monday.at("09:00").do(job)
    
    # 或者每天运行
    # schedule.every().day.at("09:00").do(job)
    
    print("GEO 监测服务已启动...")
    while True:
        schedule.run_pending()
        time.sleep(60)
```

**使用方法**：
1. 安装依赖：`pip install openai anthropic schedule`
2. 填入 API Key
3. 运行：`python geo_monitor_auto.py`
4. 或者部署到服务器/云函数
